Cricket Score Board using C 

                
                    #include 

                    // Define structure for cricket score board
                    struct CricketScoreBoard {
                        int runs;
                        int wickets;
                        float overs;
                    };
                    
                    // Function to display the current score board
                    void displayScoreBoard(struct CricketScoreBoard scoreBoard) {
                        printf("\nCurrent Score Board:\n");
                        printf("Runs: %d\n", scoreBoard.runs);
                        printf("Wickets: %d\n", scoreBoard.wickets);
                        printf("Overs: %.1f\n", scoreBoard.overs);
                    }
                    
                    // Function to update the score board based on runs, wickets, and overs
                    void updateScoreBoard(struct CricketScoreBoard *scoreBoard, int runs, int wickets, float overs) {
                        scoreBoard->runs += runs;
                        scoreBoard->wickets += wickets;
                        scoreBoard->overs = overs;
                    }
                    
                    int main() {
                        struct CricketScoreBoard scoreBoard = {0, 0, 0.0};
                        int runs, wickets;
                        float overs;
                    
                        printf("Cricket Score Board\n");
                    
                        // Update the score board based on user input
                        printf("Enter runs scored in this over: ");
                        scanf("%d", &runs);
                    
                        printf("Enter wickets fallen in this over: ");
                        scanf("%d", &wickets);
                    
                        printf("Enter overs bowled: ");
                        scanf("%f", &overs);
                    
                        updateScoreBoard(&scoreBoard, runs, wickets, overs);
                    
                        // Display the updated score board
                        displayScoreBoard(scoreBoard);
                    
                        return 0;
                    }
