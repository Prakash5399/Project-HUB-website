                                padding: 10px;
                                margin: 10px;
                                cursor: pointer;
                                }
                            </style>
                        </head>
                        <body>
                            <h2>Binary Calculator</h2>
                            <input type="text" id="binaryInput" placeholder="Enter binary number">
                            <br>
                            <button onclick="convertToDecimal()">Convert to Decimal</button>
                            <button onclick="convertToBinary()">Convert to Binary</button>
                            <br>
                            <p id="result"></p>

                        <script>
                            function convertToDecimal() {
                            const binaryInput = document.getElementById('binaryInput').value;
                            const decimalResult = parseInt(binaryInput, 2);
                            document.getElementById('result').innerHTML = 'Decimal: ' + decimalResult;
                            }

                            function convertToBinary() {
                            const decimalInput = document.getElementById('binaryInput').value;
                            const binaryResult = (parseInt(decimalInput)).toString(2);
                            document.getElementById('result').innerHTML = 'Binary: ' + binaryResult;
                            }
                        </script>
                        </body>
