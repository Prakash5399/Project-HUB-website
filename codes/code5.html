School Billing System using C 

                
                    #include 
                    #include 
                    
                    // Define structure for student
                    struct Student {
                        char name[50];
                        int rollNumber;
                        float feesPaid;
                    };
                    
                    // Function to display student details
                    void displayStudent(struct Student student) {
                        printf("\nStudent Details:\n");
                        printf("Name: %s\n", student.name);
                        printf("Roll Number: %d\n", student.rollNumber);
                        printf("Fees Paid: %.2f\n", student.feesPaid);
                    }
                    
                    // Function to calculate remaining fees
                    float calculateRemainingFees(float totalFees, float feesPaid) {
                        return totalFees - feesPaid;
                    }
                    
                    // Function to process billing
                    void processBilling(struct Student *student, float totalFees) {
                        float feesToPay;
                    
                        printf("Enter the amount to pay: ");
                        scanf("%f", &feesToPay);
                    
                        if (feesToPay <= 0) {
                            printf("Invalid amount. Amount should be greater than 0.\n");
                            return;
                        }
                    
                        student->feesPaid += feesToPay;
                    
                        printf("Payment successful!\n");
                        printf("Remaining Fees: %.2f\n", calculateRemainingFees(totalFees, student->feesPaid));
                    }
                    
                    int main() {
                        struct Student student1 = {"John Doe", 101, 0.0};
                        float totalFees = 1000.0; // Set the total fees for the student
                    
                        printf("School Billing System\n");
                    
                        displayStudent(student1);
                    
                        // Process billing for the student
                        processBilling(&student1, totalFees);
                    
                        // Display updated student details
                        displayStudent(student1);
                    
                        return 0;
                    }
