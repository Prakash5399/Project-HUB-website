                    // Simplified Order class
                    class Order {
                    public:
                        std::string symbol;
                        int quantity;
                        double price;
                        std::string side; // "Buy" or "Sell"
                        std::time_t timestamp;
                    
                        Order(const std::string& symbol, int quantity, double price, const std::string& side)
                            : symbol(symbol), quantity(quantity), price(price), side(side) {
                            timestamp = std::time(nullptr);
                        }
                    };
                    
                    // Simplified OrderBook class
                    class OrderBook {
                    private:
                        std::vector orders;
                    
                    public:
                        // Add an order to the order book
                        void addOrder(const Order& order) {
                            orders.push_back(order);
                            std::cout << "Order added: " << order.side << " " << order.quantity << " shares of " << order.symbol << " at $" << order.price << std::endl;
                        }
                    
                        // Display the current orders in the order book
                        void displayOrderBook() const {
                            std::cout << "\nOrder Book:\n";
                            for (const auto& order : orders) {
                                std::cout << "Symbol: " << order.symbol << " | Quantity: " << order.quantity << " | Price: $" << order.price
                                            << " | Side: " << order.side << " | Timestamp: " << std::asctime(std::localtime(&order.timestamp));
                            }
                        }
                    };
                    
                    int main() {
                        OrderBook orderBook;
                    
                        // Sample order creation
                        Order buyOrder("AAPL", 10, 150.0, "Buy");
                        Order sellOrder("AAPL", 5, 155.0, "Sell");
                    
                        // Add orders to the order book
                        orderBook.addOrder(buyOrder);
                        orderBook.addOrder(sellOrder);
                    
                        // Display the order book
                        orderBook.displayOrderBook();
                    
                        return 0;
                    }
