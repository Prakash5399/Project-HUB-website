Task Sheduler using JavaScript 

                
                    <!DOCTYPE html>
                    <html lang="en">
                        <head>
                            <meta charset="UTF-8">
                            <meta name="viewport" content="width=device-width, initial-scale=1.0">
                            <title>Task Scheduler</title>
                            <style>
                                body {
                                font-family: Arial, sans-serif;
                                text-align: center;
                                margin: 50px;
                                }
                                input, select, button {
                                padding: 10px;
                                margin: 10px;
                                }
                                ul {
                                list-style-type: none;
                                padding: 0;
                                }
                                li {
                                border: 1px solid #ddd;
                                padding: 10px;
                                margin: 5px;
                                display: flex;
                                justify-content: space-between;
                                }
                            </style>
                        </head>
                        <body>
                            <h2>Task Scheduler</h2>
                            <label for="task">Task:</label>
                            <input type="text" id="task" placeholder="Enter task">
                            <br>
                            <label for="date">Date:</label>
                            <input type="date" id="date">
                            <br>
                            <label for="time">Time:</label>
                            <input type="time" id="time">
                            <br>
                            <button onclick="scheduleTask()">Schedule Task</button>
                            <br>
                            <h3>Scheduled Tasks</h3>
                            <ul id="taskList"></ul>

                        <script>
                            function scheduleTask() {
                            const taskInput = document.getElementById('task').value;
                            const dateInput = document.getElementById('date').value;
                            const timeInput = document.getElementById('time').value;

                            if (taskInput && dateInput && timeInput) {
                                const scheduledTime = new Date(`${dateInput}T${timeInput}`);
                                const currentTime = new Date();

                                if (scheduledTime > currentTime) {
                                const taskList = document.getElementById('taskList');
                                const listItem = document.createElement('li');
                                listItem.innerHTML = `
                                    ${taskInput}
                                    ${scheduledTime.toLocaleString()}
                                    
                                `;
                                taskList.appendChild(listItem);

                                // Clear input fields
                                document.getElementById('task').value = '';
                                document.getElementById('date').value = '';
                                document.getElementById('time').value = '';
                                } else {
                                alert('Please select a future date and time for the task.');
                                }
                            } else {
                                alert('Please fill in all the fields.');
                            }
                            }

                            function removeTask(button) {
                            const listItem = button.parentNode;
                            listItem.parentNode.removeChild(listItem);
                            }
                        </script>
                        </body>
