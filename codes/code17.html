Monty Hall Simulation using Python 

                
                    import random

                    def monty_hall_simulation(num_simulations):
                        stay_wins = 0
                        switch_wins = 0

                        for _ in range(num_simulations):
                            # Initialize doors with a car behind one and goats behind the others
                            doors = ['goat'] * 3
                            car_index = random.randint(0, 2)
                            doors[car_index] = 'car'

                            # Contestant makes initial choice
                            contestant_choice = random.randint(0, 2)

                            # Monty opens a door with a goat behind it
                            remaining_doors = [i for i in range(3) if i != contestant_choice and doors[i] == 'goat']
                            monty_opens = random.choice(remaining_doors)

                            # Contestant chooses to stay or switch
                            stay_wins += doors[contestant_choice] == 'car'
                            switch_wins += doors[3 - contestant_choice - monty_opens] == 'car'

                        stay_win_percentage = (stay_wins / num_simulations) * 100
                        switch_win_percentage = (switch_wins / num_simulations) * 100

                        print(f"Stay wins: {stay_wins} ({stay_win_percentage:.2f}%)")
                        print(f"Switch wins: {switch_wins} ({switch_win_percentage:.2f}%)")

                    if __name__ == "__main__":
                        num_simulations = int(input("Enter the number of simulations: "))
                        monty_hall_simulation(num_simulations)
