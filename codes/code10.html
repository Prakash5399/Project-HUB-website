                    bool isValidCreditCard(const std::string& cardNumber) {
                        int sum = 0;
                        bool doubleDigit = false;
                    
                        for (int i = cardNumber.length() - 1; i >= 0; --i) {
                            int digit = cardNumber[i] - '0';
                    
                            if (doubleDigit) {
                                digit *= 2;
                                if (digit > 9) {
                                    digit = digit % 10 + 1;
                                }
                            }
                    
                            sum += digit;
                            doubleDigit = !doubleDigit;
                        }
                    
                        return (sum % 10 == 0);
                    }
                    
                    int main() {
                        std::string cardNumber;
                    
                        // Get the credit card number from the user
                        std::cout << "Enter your credit card number: ";
                        std::getline(std::cin, cardNumber);
                    
                        // Remove spaces from the input
                        cardNumber.erase(std::remove(cardNumber.begin(), cardNumber.end(), ' '), cardNumber.end());
                    
                        // Validate the credit card number
                        if (isValidCreditCard(cardNumber)) {
                            std::cout << "Credit card number is valid." << std::endl;
                        } else {
                            std::cout << "Credit card number is not valid." << std::endl;
                        }
                    
                        return 0;
                    }
