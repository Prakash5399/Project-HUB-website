Online Survey System using Java 

                
                    import java.util.ArrayList;
                    import java.util.List;
                    import java.util.Scanner;

                    class Question {
                        private String text;
                        private List options;

                        public Question(String text, List options) {
                            this.text = text;
                            this.options = options;
                        }

                        public String getText() {
                            return text;
                        }

                        public List getOptions() {
                            return options;
                        }
                    }

                    class Survey {
                        private List questions;

                        public Survey(List questions) {
                            this.questions = questions;
                        }

                        public void conductSurvey() {
                            Scanner scanner = new Scanner(System.in);
                            List responses = new ArrayList<>();

                            System.out.println("Welcome to the Online Survey!");

                            for (Question question : questions) {
                                System.out.println("\n" + question.getText());
                                List options = question.getOptions();
                                for (int i = 0; i < options.size(); i++) {
                                    System.out.println((i + 1) + ". " + options.get(i));
                                }

                                int selectedOption;
                                do {
                                    System.out.print("Select an option (1-" + options.size() + "): ");
                                    selectedOption = scanner.nextInt();
                                } while (selectedOption < 1 || selectedOption > options.size());

                                responses.add(options.get(selectedOption - 1));
                            }

                            System.out.println("\nThank you for completing the survey!");

                            // Display responses (you can save them to a database or file in a real-world application)
                            System.out.println("\nSurvey Responses:");
                            for (int i = 0; i < questions.size(); i++) {
                                System.out.println(questions.get(i).getText() + ": " + responses.get(i));
                            }
                        }
                    }

                    public class OnlineSurveySystem {
                        public static void main(String[] args) {
                            // Create survey questions
                            List questions = new ArrayList<>();
                            questions.add(new Question("What is your favorite programming language?",
                                    List.of("Java", "Python", "C++", "JavaScript")));
                            questions.add(new Question("How often do you code?",
                                    List.of("Daily", "Weekly", "Monthly", "Rarely")));

                            // Create and conduct the survey
                            Survey survey = new Survey(questions);
                            survey.conductSurvey();
                        }
                    }
